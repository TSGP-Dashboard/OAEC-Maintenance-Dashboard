function openEditModal(weekStr, taskNo) { const taskList = database.schedules[weekStr] || []; const item = taskList.find(t => t.no === taskNo); if (!item) return; document.getElementById('edit-week').value = weekStr; document.getElementById('edit-no').value = taskNo; document.getElementById('edit-row-id').value = item.id || ""; document.getElementById('edit-equip').value = item.equip; document.getElementById('edit-type').value = item.type; document.getElementById('edit-planner').value = item.planner; document.getElementById('edit-status').value = item.status; document.getElementById('edit-activity').value = item.activity; document.getElementById('edit-date').value = item.date; document.getElementById('edit-duration').value = item.duration; document.getElementById('edit-loc').value = item.loc; document.getElementById('edit-info').value = item.info; document.getElementById('edit-remarks').value = item.remarks || ""; // NEW: Populate the new tracking UI fields dynamically document.getElementById('edit-done-date').value = item.doneDate || ""; document.getElementById('edit-carry-date').value = item.carryOverDate || ""; toggleRemarksFieldVisibility(item.status); document.getElementById('edit-modal').classList.remove('hidden'); } function handleEditFormSubmit(event) { event.preventDefault(); const targetIdValue = encodeURIComponent(document.getElementById('edit-row-id').value); if (!targetIdValue) { alert("Missing row ID mapping index."); return; } const selectedStatus = document.getElementById('edit-status').value; const todayStamp = new Date().toISOString().split('T')[0]; // Retain old values if they exist, or set them based on status changes let doneDateValue = document.getElementById('edit-done-date').value; let carryOverDateValue = document.getElementById('edit-carry-date').value; if (selectedStatus === 'DONE') { doneDateValue = doneDateValue || todayStamp; carryOverDateValue = ""; // clear out if it's finally finished } else if (selectedStatus === 'NOT DONE') { carryOverDateValue = carryOverDateValue || todayStamp; doneDateValue = ""; // clear out since it's incomplete } else { doneDateValue = ""; carryOverDateValue = ""; } const updatedData = { "Equipment No": document.getElementById('edit-equip').value.toUpperCase(), "Type": document.getElementById('edit-type').value, "Assigned Planner": document.getElementById('edit-planner').value, "Current Status": selectedStatus, "Activity Description": document.getElementById('edit-activity').value, "Target Date": document.getElementById('edit-date').value, "Duration": document.getElementById('edit-duration').value, "Location": document.getElementById('edit-loc').value, "Parts Status": document.getElementById('edit-info').value, "DONE DATE": doneDateValue, "Carry-over date": carryOverDateValue, "REASON FOR CARRY OVER": document.getElementById('edit-remarks').value }; fetch(`${SHEETDB_API_URL}/ID/${targetIdValue}`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ data: updatedData }) }) .then(res => { if(!res.ok) throw new Error(); closeEditModal(); loadDatabaseFromCloud(); }) .catch(() => alert("Error saving data changes.")); }
